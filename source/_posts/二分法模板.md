#### 寻找左边第一个x
```c++
int l = 0, r = n - 1;
while (l < r) {
    int mid = l + r >> 1;
    if (arr[mid] < x) l = mid + 1;
    else r = mid;
}
if (arr[l] != x) return -1;
return l;  // l和r都是一样的，因为l = r跳出循环
```

#### 寻找右边最后一个x

```c++
// 寻找最后一个x
int l = 0, r = n - 1;
while (l < r) {
    int mid = l + r + 1 >> 1; (注意改边界问题)
    if (arr[mid] > x) r = mid - 1;
    else l = mid;
}
if (arr[l] != x) return -1;
return l;
```
